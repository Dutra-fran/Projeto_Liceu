CREATE TABLE Cadastro (
ID INT PRIMARY KEY AUTO_INCREMENT,
Nome VARCHAR(300) NOT NULL,
Email VARCHAR(300) NOT NULL UNIQUE,
Senha VARCHAR(32) NOT NULL,
ID_Sala SMALLINT NOT NULL,
ID_Exatas SMALLINT NOT NULL,
ID_Humanas SMALLINT NOT NULL,
FOREIGN KEY(ID_Sala) REFERENCES Salas(ID_Sala),
FOREIGN KEY(ID_Exatas) REFERENCES Exatas(ID_Exatas),
FOREIGN KEY(ID_Humanas) REFERENCES Humanas(ID_Humanas)
);

CREATE TABLE Professores (
ID_Professor SMALLINT PRIMARY KEY AUTO_INCREMENT,
Nome VARCHAR(300) NOT NULL,
Email VARCHAR(300) NOT NULL UNIQUE,
Senha VARCHAR(32) NOT NULL
);

CREATE TABLE Exatas (
ID_Exatas SMALLINT PRIMARY KEY AUTO_INCREMENT,
ID_Professor SMALLINT NOT NULL,
Disciplina VARCHAR(50) NOT NULL,
FOREIGN KEY(ID_Professor) REFERENCES Professores(ID_Professor)
);

CREATE TABLE Humanas (
ID_Humanas SMALLINT PRIMARY KEY AUTO_INCREMENT,
ID_Professor SMALLINT NOT NULL,
Disciplina VARCHAR(50) NOT NULL,
FOREIGN KEY(ID_Professor) REFERENCES Professores(ID_Professor)
);

CREATE TABLE Salas_Prof (
ID SMALLINT PRIMARY KEY AUTO_INCREMENT,
ID_Sala SMALLINT NOT NULL,
ID_Professor SMALLINT NOT NULL,
FOREIGN KEY(ID_Sala) REFERENCES Salas(ID_Sala),
FOREIGN KEY(ID_Professor) REFERENCES Professores(ID_Professor)
);

CREATE TABLE Verifica_Prova (
ID SMALLINT PRIMARY KEY AUTO_INCREMENT,
ID_Cadastro INT NOT NULL,
ID_Sala SMALLINT NOT NULL,
Disciplina VARCHAR(50) NOT NULL,
FOREIGN KEY(ID_Cadastro) REFERENCES Cadastro(ID),
FOREIGN KEY(ID_Sala) REFERENCES Salas(ID_Sala)
);

CREATE TABLE Resposta_Aluno (
ID INT PRIMARY KEY AUTO_INCREMENT,
ID_Sala SMALLINT NOT NULL,
ID_Cadastro INT NOT NULL,
Questao CHAR(1) NOT NULL,
Disciplina VARCHAR(50) NOT NULL,
FOREIGN KEY(ID_Sala) REFERENCES Salas(ID_Sala),
FOREING KEY(ID_Cadastro) REFERENCES Cadastro(ID)
);
